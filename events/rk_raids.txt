namespace = raid_event

# TOB
raid_event.0001 = {
	hidden = yes
	scope = none
	

	immediate = {
		mp_pause_travel_plan = yes
		play_music_cue = rk_cue_vampfight
		create_character = {
			location = title:c_vers.holder.capital_province
			template = accolade_huntsmaster_character
			name = "Verzik Vitur"
			gender = female
			culture = culture:vampyre
			dynasty = none
			faith = faith:zamorak
			trait = vampire_character
			save_scope_as = fighter_mc_fighter_chap
		}
		configure_start_single_combat_effect = {
			SC_INITIATOR = root
			SC_ATTACKER = root
			SC_DEFENDER = scope:fighter_mc_fighter_chap
			FATALITY = possible
			FIXED = no
			LOCALE = throne_room
			INVALIDATION_EVENT = single_combat.1006
			OUTPUT_EVENT = raid_event.0100
		}
	}

}

raid_event.0100 = {
	hidden = yes
	scope = none

	immediate = {
		
		current_travel_plan ?= {
			if = {
				limit = { is_paused = yes }
				resume_travel_plan = yes 
			}
		}
		if = {
			limit = { 
				scope:activity = {
					has_activity_option = { 
						category = special_type 
						option = raid_tob
					}
				}
			}
			trigger_event = { 
				id = raid_event.0101
				days = 1
			}
		}
		if = {
			limit = { 
				scope:activity = {
					has_activity_option = { 
						category = special_type 
						option = raid_tombs
					}
				}
			}
			trigger_event = { 
				id = raid_event.0102
				days = 1
			}
		}
		if = {
			limit = { 
				scope:activity = {
					has_activity_option = { 
						category = special_type 
						option = raid_xeric
					}
				}
			}
			trigger_event = { 
				id = raid_event.0103
				days = 1
			}
		}
		
		if = {
			limit = { has_character_flag = raid_character_flag }
			remove_character_flag = raid_character_flag
		}

	}

	after = {
		mp_resume_travel_plan = yes
	}

}
