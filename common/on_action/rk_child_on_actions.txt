
on_birthday = {
    on_actions = {
		birthday_wizard
	}
}

birthday_wizard = {
	trigger = { age = 16 }

	effect = {
		if = {
			limit = {
				OR = {
					root.employer = { has_primary_title = title:c_wizard }
					root.employer = { has_primary_title = title:d_wizard }
				}
			}
			give_nickname = nick_wizard
			add_character_flag = {
				flag = wizard
			}
		}
	}
}
