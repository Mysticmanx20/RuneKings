on_game_start_after_lobby = {
	effect = {

		
		#starting cultural acceptance
		culture:vampyre = {
			change_cultural_acceptance = {
				target = culture:werewolf
				value = 85
				desc = cultural_acceptance_historical_relations
			}
		}

		culture:briton = {
			change_cultural_acceptance = {
				target = culture:kandarin
				value = 25
				desc = cultural_acceptance_historical_relations
			}
		}

		culture:fremennik = {
			change_cultural_acceptance = {
				target = culture:moonclan
				value = -50
				desc = cultural_acceptance_historical_relations
			}
			change_cultural_acceptance = {
				target = culture:oldfremennik
				value = -50
				desc = cultural_acceptance_historical_relations
			}
		}

		culture:moonclan = {
			change_cultural_acceptance = {
				target = culture:oldfremennik
				value = -50
				desc = cultural_acceptance_historical_relations
			}
		}

		### CE1 LEGITIMACY SETUP ###
		every_ruler = {
			limit = {
				exists = dynasty
				highest_held_title_tier >= tier_county
				# Exclude unplayable for now
				NOR = {
					government_has_flag = government_is_republic
					government_has_flag = government_is_theocracy
				}
				is_landed = yes
			}
			add_legitimacy = base_legitimacy_value
		}
	}
}


########RuneKings

vamps = {
	effect = {
		every_ruler = {
			limit = {
				culture = culture:vampyre
				NOT = { has_trait = cannibal }
			}
			add_trait = cannibal
		}
	}
}

vampyres = {
	effect = {
		every_living_character = {
			limit = {
				culture = culture:vampyre
				NOT = { has_trait = vampire_character }
			}
			add_trait = vampire_character
		}
	}
}

start_religion = {
	effect = {
		if = {
			limit = { exists = title:k_asgarnia.holder }
			title:k_asgarnia.holder = {
				every_vassal = {
					limit = {
						has_primary_title = title:c_taverley
					}
					vassal_contract_set_obligation_level = { type = religious_rights level = 1 }
				}
			}
		}
	}
}

start_whiteknights = {

	effect = {
		if = {
			limit = { exists = title:k_asgarnia.holder }

			create_holy_order_neutral_effect = {
				LEADER = character:900305
				CAPITAL = title:b_whiteknight
				NEW_HO_SCOPE = new_holy_order
				FOUNDER = character:900301
			}

			scope:new_holy_order = {
                copy_title_history = title:d_whiteknights
            }
			
		}

	}
}

wizards = {
	effect = {
		every_ruler = {
			limit = {
				OR = {
					has_primary_title = title:c_wizard
					has_primary_title = title:d_wizard
				}
			}
			every_courtier = {
				limit = {
					NOT = { has_council_position = councillor_court_chaplain }
				}
				give_nickname = nick_wizard
				add_character_flag = {
				flag = wizard
				}
			}
		}
	}
}

whiteknights = {
	effect = {
		every_ruler = {
			limit = {
				OR = {
					has_primary_title = title:k_asgarnia
					has_primary_title = title:d_falador
				}
			}
			every_knight = {
				limit = { 
					is_female = no
					is_landed = no
					has_faith = faith:saradomin
				}
				give_nickname = nick_sir
			}
			every_knight = {
				limit = { 
					is_female = yes
					is_landed = no
					has_faith = faith:saradomin
				}
				give_nickname = nick_lady
			}
		}
	}
}


on_join_court = {

	on_actions = {
		whiteknights
	}

	events = {
		court_maintenance.0005 # Add 'Order member' trait to courtiers who join holy order court
	}
	effect = {
		if = {
			limit = {
				is_adult = yes
				NOT = { has_council_position = councillor_court_chaplain } 
				scope:new_employer = {
					NOT = {is_spouse_of = root }
					OR = {
						has_primary_title = title:c_wizard
						has_primary_title = title:d_wizard
					}
				}
			}
			give_nickname = nick_wizard
			add_character_flag = {
				flag = wizard
			}
		}
	}
}
