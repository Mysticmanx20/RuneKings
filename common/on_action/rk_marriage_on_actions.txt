
on_marriage = {
	on_actions = {
		rk_marriage_nick 
	}
}

rk_marriage_nick = {
    effect = {

		if = {
			limit = {
				OR = {
					scope:spouse = { has_nickname = nick_wizard }
					scope:spouse = { has_nickname = nick_sir }
					scope:spouse = { has_nickname = nick_lady }
				}
				root = { is_landed = yes }
			}
			scope:spouse = { remove_nickname = yes }
		}
	}
}
