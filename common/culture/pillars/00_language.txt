
### RuneKings

language_common = {
	type = language
	is_shown = {
		language_is_shown_trigger = {
			LANGUAGE = language_common
		}
	}
	ai_will_do = {
		value = 10
		if = {
			limit = { has_cultural_pillar = language_common }
			multiply = 10
		}
	}
	
	color = frankish
}
